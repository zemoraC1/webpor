Autenticación y Middleware

JWT Middleware
JavaScript 
// middleware/auth.js
const jwt = require('jsonwebtoken');

const authenticate = async (req, res, next) => {
  const token = req.headers.authorization
    ?.replace('Bearer ', '');

  if (!token) return res.status(401).json({
    error: 'Token requerido'
  });

  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET);
    req.user = await getUserById(payload.sub);
    if (req.user.status !== 'active')
      return res.status(403).json({ error: 'Cuenta suspendida' });
    next();
  } catch (err) {
    res.status(401).json({ error: 'Token inválido' });
  }
};

const requireRole = (...roles) => (req, res, next) => {
  if (!roles.includes(req.user.role))
    return res.status(403).json({ error: 'Sin permisos' });
  next();
};

module.exports = { authenticate, requireRole };

// workers/videoProcessor.js
const { Worker } = require('bullmq');
const ffmpeg = require('fluent-ffmpeg');

const videoWorker = new Worker('video-processing', async (job) => {
  const { videoId, inputPath, outputDir } = job.data;

  // 1. Generar miniaturas
  await generateThumbnails(inputPath, outputDir, videoId);

  // 2. Transcodificar a múltiples resoluciones
  const resolutions = [
    { name: '360p',  width: 640,  height: 360,  bitrate: '800k'  },
    { name: '720p',  width: 1280, height: 720,  bitrate: '2500k' },
    { name: '1080p', width: 1920, height: 1080, bitrate: '5000k' }
  ];

  for (const res of resolutions) {
    await transcodeToHLS(inputPath, outputDir, res);
    await job.updateProgress(
      (resolutions.indexOf(res) + 1) / resolutions.length * 100
    );
  }

  // 3. Generar manifiesto HLS maestro
  await generateMasterManifest(outputDir, videoId, resolutions);

  // 4. Generar preview clip (5 segundos)
  await generatePreviewClip(inputPath, outputDir, videoId);

  // 5. Subir todo a R2
  const r2Keys = await uploadToR2(outputDir, videoId);

  // 6. Actualizar BD
  await updateVideoRecord(videoId, {
    status: 'active',
    ...r2Keys
  });
}, { concurrency: 3 });

async function transcodeToHLS(input, outputDir, res) {
  return new Promise((resolve, reject) => {
    ffmpeg(input)
      .videoCodec('libx264')
      .audioCodec('aac')
      .size(`${res.width}x${res.height}`)
      .videoBitrate(res.bitrate)
      .outputOptions([
        '-hls_time 6',
        '-hls_playlist_type vod',
        '-hls_segment_filename',
        `${outputDir}/${res.name}_%03d.ts`
      ])
      .output(`${outputDir}/${res.name}.m3u8`)
      .on('end', resolve)
      .on('error', reject)
      .run();
  });
}

Reproductor HLS (Next.js Component)
React / Next.js 
// components/player/VideoPlayer.jsx
'use client';
import { useEffect, useRef } from 'react';
import Hls from 'hls.js';

export default function VideoPlayer({ videoId, hlsUrl }) {
  const videoRef = useRef(null);

  useEffect(() => {
    const video = videoRef.current;
    if (!video || !hlsUrl) return;

    if (Hls.isSupported()) {
      const hls = new Hls({
        enableWorker: true,
        lowLatencyMode: false,
        xhrSetup: (xhr) => {
          xhr.setRequestHeader(
            'Authorization',
            `Bearer ${getToken()}`
          );
        }
      });
      hls.loadSource(hlsUrl);
      hls.attachMedia(video);
    } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
      video.src = hlsUrl; // Safari nativo
    }
  }, [hlsUrl]);

  return (
    <video
      ref={videoRef}
      controls
      playsInline
      className="w-full aspect-video rounded-xl"
    />
  );
}

// Signed URLs para R2 (anti-hotlink)
async function getSignedStreamUrl(videoId, userId) {
  const hasAccess = await checkVideoAccess(videoId, userId);
  if (!hasAccess) throw new ForbiddenError();

  const key = `videos/${videoId}/master.m3u8`;

  const signedUrl = await r2Client.getSignedUrl(
    new GetObjectCommand({ Bucket: R2_BUCKET, Key: key }),
    { expiresIn: 3600 }  // 1 hora
  );

  return signedUrl;
}

// Endpoint de streaming protegido

Revenue Sharing — Fórmula de Distribución
JavaScript 
// services/revenueShare.js
const PLATFORM_CUT = 0.25;   // 25% plataforma
const CREATOR_CUT  = 0.75;   // 75% creador

async function distributeRevenue(transaction) {
  const { amount, videoId, userId } = transaction;

  const video = await getVideo(videoId);
  const creatorShare = amount * CREATOR_CUT;
  const platformShare = amount * PLATFORM_CUT;

  // Acumular en wallet del creador
  await addToCreatorWallet(video.user_id, creatorShare);

  // Registrar distribución
  await logRevenueDistribution({
    transaction_id: transaction.id,
    creator_id: video.user_id,
    creator_amount: creatorShare,
    platform_amount: platformShare,
    currency: transaction.currency
  });

  // Factura automática si supera umbral
  if (await walletExceedsThreshold(video.user_id)) {
    await scheduleCreatorPayout(video.user_id);
  }
}
